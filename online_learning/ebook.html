const mongoose = require('mongoose'); const UserSchema = new mongoose.Schema({ fullName: { type: String, required: true }, email: { type: String, required: true, unique: true }, password: { type: String, required: true }, likedUniversities: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Institution' }], appliedUniversities: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Institution' }], }); const User = mongoose.model('User', UserSchema); module.exports = User;
